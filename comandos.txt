-> configura o pc com nome e e-mail do responsavel pelo codigo
git config --global user.name "NOME AQUI"
git config --global user.email "EMAIL AQUI"

-> navegando no cmd
cd nome-da-pasta // entra em uma pasta
cd .. // volta uma pasta
cd /    // vai para o C:

-> inicializa um repositorio git(utilizado apenas uma vez)
git init

-> verificar o status do repositorio
git status

-> adicionar os novos arquivos no git para versionar
git add .

-> salva uma versão dos arquivos
git commit -m "Porque estou salvando???"

-> envia informação do computador para o github
git push

-> baixa a atualização dos arquivos do github para uma pasta já existente
git pull
